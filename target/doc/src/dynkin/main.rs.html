1// Count the number of Dynkin systems on sets of sizes 1, ..., n.
2// A Dynkin system on a set X is a collection D of subsets of X that satisfies
3//      (i) ∅ ∈ D,
4//     (ii) if A ∈ D, then X-A ∈ D,
5//    (iii) if A1, A2, ... ∈ D are disjoint, then ⋃ Ai ∈ D.
6//
7// This implementation mostly follows the approach used by the Mathoverflow user Peter Taylor
8// in the post linked below:
9// https://mathoverflow.net/questions/405736/what-is-the-number-of-finite-dynkin-systems?rq=1
10//
11// His python implementation calculates up to n=7 in about 40 seconds on my machine. This
12// rust implementation calculates up to n=7 in about 0.15 seconds on my machine.
13
14const MAX_N: usize = 7; // n=7 takes <1 second, n=8 takes ~24 hours
15const MAX_SUBSETS: usize = 1 << MAX_N; // The cardinality of the powerset of a set of size MAX_N
16const BITSET_WORDS: usize = MAX_SUBSETS / 64;
17
18/// A bitwise operable representation of the powerset, where 1 represents a valid Dynkin system.
19type Bitset = [u64; BITSET_WORDS];
20
21/// Sets all bits in the bitset to zero.
22fn bs_clear(bitset: &mut Bitset) {
23    for word in bitset.iter_mut() {
24        *word = 0;
25    }
26}
27
28/// Copies the contents of one bitset to another.
29fn bs_copy(destination: &mut Bitset, source: &Bitset) {
30    destination.copy_from_slice(source);
31}
32
33/// Sets a specific bit in the bitset to 1.
34fn bs_set(bitset: &mut Bitset, index: usize) {
35    let word: usize = index >> 6;
36    let bit: usize = index & 63;
37    bitset[word] |= 1 << bit;
38}
39
40/// Gets the value of a specific bit in the bitset.
41/// Returns true if the bit is set, false otherwise.
42fn bs_get(bitset: &Bitset, index: usize) -> bool {
43    let word: usize = index >> 6;
44    let bit: usize = index & 63;
45    ((bitset[word] >> bit) & 1) != 0
46}
47
48/// A queue to manage the elements being processed during the closure extension.
49struct Queue {
50    data: [usize; MAX_SUBSETS],
51    len: usize,
52}
53
54impl Queue {
55    fn clear(&mut self) {
56        self.len = 0;
57    }
58
59    fn push(&mut self, x: usize) {
60        self.data[self.len] = x;
61        self.len += 1;
62    }
63}
64
65/// Extends the closure by including a new element and updating the closure set.
66/// Returns false if the extension is not valid (i.e., it conflicts with excluded elements).
67/// Parameters:
68/// - `omega`: the maximum element in the set (i.e., the size of the set minus one)
69/// - `included`: the current set of included elements (the closure)
70/// - `extension`: the new element to be included in the closure
71/// - `excluded`: the current set of excluded elements (elements that cannot be included)
72/// - `closure`: the current closure set to be updated
73/// - `queue`: a queue to manage the elements being processed
74/// Returns true if the extension is valid, false otherwise.
75fn extend_closure(
76    omega: usize,
77    included: &Bitset,
78    extension: usize,
79    excluded: &Bitset,
80    closure: &mut Bitset,
81    queue: &mut Queue,
82) -> bool {
83    bs_copy(closure, included);
84    bs_set(closure, extension);
85    queue.clear();
86    queue.push(extension);
87    let mut queue_index: usize = 0;
88
89    while queue_index < queue.len {
90        let x: usize = queue.data[queue_index];
91        queue_index += 1;
92        let complement: usize = omega ^ x;
93        if !bs_get(closure, complement) {
94            if bs_get(excluded, complement) {
95                return false;
96            }
97            bs_set(closure, complement);
98            queue.push(complement);
99        }
100
101        for word in 0..BITSET_WORDS {
102            let mut bits: u64 = closure[word];
103
104            while bits != 0 {
105                let lsb: usize = bits.trailing_zeros() as usize;
106                bits &= bits - 1;
107                let y: usize = (word << 6) + lsb; // Calculates the index of the bit
108
109                // If x ⋂ y = ∅, then we can add x ⋃ y to the closure
110                if (x & y) == 0 {
111                    let z: usize = x | y;
112                    if !bs_get(closure, z) {
113                        if bs_get(excluded, z) {
114                            return false;
115                        }
116                        bs_set(closure, z);
117                        queue.push(z);
118                    }
119                }
120            }
121        }
122    }
123    true
124}

126/// Recursively counts the number of valid subsets of the closure.
127/// Given a set system F and a property P, the closure of F with respect to P is the smallest superset of F satisfying P.
128/// Parameters:
129/// - `omega`: the maximum element in the set (i.e., the size of the set minus one)
130/// - `lower_bound`: the lower bound for the next element to consider
131/// - `included`: the current set of included elements (the closure)
132/// - `excluded`: the current set of excluded elements (elements that cannot be included)
133/// - `queue`: a queue to manage the elements being processed
134/// Returns the count of valid subsets of the closure.
135fn inner(
136    omega: usize,
137    lower_bound: usize,
138    included: &Bitset,
139    excluded: &mut Bitset,
140    _queue: &mut Queue,
141    depth: usize,
142) -> usize {
143    let mut count: usize = 1;
144    let limit: usize = (omega + 1) >> 1;
145
146    // local queue for closure extension
147    let mut queue_local: Queue = Queue {
148        data: [0; MAX_SUBSETS],
149        len: 0,
150    };
151
152    for x in lower_bound..limit {
153        if bs_get(included, x) || bs_get(excluded, x) {
154            continue;
155        }
156
157        // Inclusion branch
158        let mut closure = [0u64; BITSET_WORDS];
159        queue_local.clear();
160        if extend_closure(omega, included, x, excluded, &mut closure, &mut queue_local) {
161            let mut new_excluded = *excluded;
162            count += inner(
163                omega,
164                x + 1,
165                &closure,
166                &mut new_excluded,
167                &mut queue_local,
168                depth,
169            );
170        }
171
172        // Exclusion branch
173        bs_set(excluded, x);
174        bs_set(excluded, omega ^ x); // D-x
175    }
176
177    count
178}

180fn main() {
181    // Calculate number of Dynkin systems for each set size
182    for n in 0..=MAX_N {
183        let omega: usize = if n > 0 { (1 << n) - 1 } else { 0 };
184
185        // Initial included bitset: {∅, X}
186        let mut included = [0u64; BITSET_WORDS];
187        bs_clear(&mut included);
188        bs_set(&mut included, 0);
189        bs_set(&mut included, omega);
190
191        // Prepare root_excluded array up to halfway
192        let limit = (omega + 1) >> 1;
193        let mut root_excluded = vec![[0u64; BITSET_WORDS]; limit];
194        for m in 1..limit {
195            root_excluded[m] = root_excluded[m - 1];
196            bs_set(&mut root_excluded[m], m);
197            bs_set(&mut root_excluded[m], omega ^ m);
198        }
199
200        // Parallel over m choices
201        use rayon::prelude::*;
202        let sum: usize = (1..limit)
203            .into_par_iter()
204            .map(|m| {
205                if bs_get(&included, m) || bs_get(&root_excluded[m - 1], m) {
206                    return 0;
207                }
208                // Build closure and count branch
209                let mut closure = [0u64; BITSET_WORDS];
210                let mut queue: Queue = Queue {
211                    data: [0; MAX_SUBSETS],
212                    len: 0,
213                };
214                let ex_here = &root_excluded[m - 1];
215                let mut count: usize = 0;
216                if extend_closure(omega, &included, m, ex_here, &mut closure, &mut queue) {
217                    let mut new_exc = *ex_here;
218                    bs_set(&mut new_exc, m);
219                    bs_set(&mut new_exc, omega ^ m);
220                    count += inner(omega, m + 1, &closure, &mut new_exc, &mut queue, 0);
221                }
222                count
223            })
224            .sum();
225
226        let total: usize = sum + 1;
227        println!("{} -> {}", n, total);
228    }
